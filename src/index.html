Create a podlet (page fragment) and load it in a layout (page)

âœï¸ podlet.js

const express = require('express');
const Podlet = require('@podium/podlet');

const app = express();

const podlet = new Podlet({
  name: 'myPodlet',
  version: '1.0.0',
});

app.get(podlet.manifest(), (req, res) => {
  res.send(podlet);
});

app.get(podlet.content(), (req, res) => res.send(`
  <div>Podlet content</div>
`));

app.listen(7200);

ğŸ›  layout.js

const express = require('express');
const Layout = require('@podium/layout');

const layout = new Layout('myLayout');
const podlet = layout.client.register({
    name: 'myPodlet',
    url: 'http://localhost:7000/manifest.json'
});

const app = express();
app.use(layout.middleware());

app.get('/', async (req, res) => {
    const content = await podlet.fetch();
    res.send(`
        <html><body>${content}</body></html>
    `);
});

app.listen(7000);
